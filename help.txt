ZTerminal can be confusing at times, so here is a guide.
--==--==--==--==--==--==--==--==--==--==--==--==--==--
print - repeats a string back
	syntax:
	print-string

version - tells you your current zt version
	syntax:
	version
	or
	ver

wv - tells you your current windows version
	syntax:
	wv

ndir - creates a directory
	syntax:
	ndir-dirname

tofile - puts content into a file
	syntax:
	tofile-path\filename-content

time - tells the time
	syntax:
	time

wincmd - runs wcp commands
	syntax:
	wincmd-command arg

clear - clears the screen
	syntax:
	clear

pause - pauses until a key is pressed
	syntax:
	pause

exit - closes ZTerminal
	syntax:
	exit

ping - pings an ip address or website
	syntax:
	ping-ip

title - changes the window title
	syntax:
	title-text

call - calls a .zt file
	syntax:
	call-file

delf - deletes a file
	syntax:
	delf-path\file

deldir - deletes a directory
	syntax:
	deldir-path

start - starts a program
	syntax:
	start-program
